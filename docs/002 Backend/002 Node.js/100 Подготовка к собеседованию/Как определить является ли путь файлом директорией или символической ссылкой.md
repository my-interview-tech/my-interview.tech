Для того чтобы определить, является ли путь файлом, директорией или символической ссылкой в Node.js, можно использовать метод `fs.stat()` (или его асинхронный аналог `fs.promises.stat()`), а затем вызвать соответствующие методы объекта `fs.Stats`, такие как `isFile()`, `isDirectory()` и `isSymbolicLink()`.

Вот пример, как это можно сделать:

### Пример с использованием `fs.stat()`:

```javascript
const fs = require("fs")

const path = "example.txt" // Укажите путь к файлу или директории

fs.stat(path, (err, stats) => {
  if (err) {
    console.error("Ошибка:", err)
  } else {
    if (stats.isFile()) {
      console.log(`${path} - Это файл`)
    } else if (stats.isDirectory()) {
      console.log(`${path} - Это директория`)
    } else if (stats.isSymbolicLink()) {
      console.log(`${path} - Это символическая ссылка`)
    } else {
      console.log(`${path} - Это неизвестный тип`)
    }
  }
})
```

### Пример с использованием `fs.promises.stat()` (асинхронная версия с `async/await`):

```javascript
const fs = require("fs").promises

const path = "example.txt" // Укажите путь к файлу или директории

async function checkPathType() {
  try {
    const stats = await fs.stat(path)

    if (stats.isFile()) {
      console.log(`${path} - Это файл`)
    } else if (stats.isDirectory()) {
      console.log(`${path} - Это директория`)
    } else if (stats.isSymbolicLink()) {
      console.log(`${path} - Это символическая ссылка`)
    } else {
      console.log(`${path} - Это неизвестный тип`)
    }
  } catch (err) {
    console.error("Ошибка:", err)
  }
}

checkPathType()
```

### Описание методов:

- **`stats.isFile()`** — возвращает `true`, если путь указывает на файл.
- **`stats.isDirectory()`** — возвращает `true`, если путь указывает на директорию.
- **`stats.isSymbolicLink()`** — возвращает `true`, если путь указывает на символическую ссылку.

Таким образом, вы можете легко определить тип пути и выполнить соответствующие действия в зависимости от его типа.
